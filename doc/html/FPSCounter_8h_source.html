00005  General purpose frames per second counter for OpenGL/GLUT GNU/Linux
00006  programs. Displays "Frames per second: N" at an arbitrary position in
00007  the window. Saves and restores the app's modelview and projection matrices,
00008  colour, and lighting.
00009 
00010  Author: Toby Howard. toby.howardy@manchester.ac.uk.
00011  Version 2.1, 3 January 2013
00012 
00013  ====================
00014 
00015  Usage: to add an on-screen frames per second counter to your program, save
00016  this file alongside your app as "frames.h", and add:
00017 
00018     #include "frames.h"
00019 
00020  immediately after all your app's other #includes; then bracket all the
00021  code in your display() function, before swapping buffers, with
00022 
00023    frameStart();
00024 
00025  and
00026 
00027    frameEnd(void *font, GLclampf r, GLclampf g, GLclampf b,
00028             float x, float y);
00029 
00030      font:    font to use, e.g., GLUT_BITMAP_HELVETICA_10
00031      r, g, b: RGB text colour
00032      x, y:    text position in window: range [0,0] (bottom left of window)
00033               to [1,1] (top right of window).
00034 
00035  ====================
00036 
00037  Example:
00038 
00039     void display(void) {
00040       glClear(GL_COLOR_BUFFER_BIT);
00041 
00042       frameStart();
00043 
00044       // all the graphics code
00045 
00046       frameEnd(GLUT_BITMAP_HELVETICA_10, 1.0, 1.0, 1.0, 0.05, 0.95);

00062 void frameEnd(void *font, GLclampf r, GLclampf g, GLclampf b,
00063               GLfloat x, GLfloat y) {
00064   /* font: font to use, e.g., GLUT_BITMAP_HELVETICA_10
00065      r, g, b: text colour
00066      x, y: text position in window: range [0,0] (bottom left of window)
00067            to [1,1] (top right of window). */
00068 
00069   float elapsedTime;
00070   char str[30];
00071   char *ch;
00072   GLint matrixMode;
00073   GLboolean lightingOn;
00074 
00075   gettimeofday(&frameEndTime, NULL);
00076 
00077   elapsedTime= frameEndTime.tv_sec - frameStartTime.tv_sec +
00078              ((frameEndTime.tv_usec - frameStartTime.tv_usec)/1.0E6);
00079   sprintf(str, "Frames per second: %2.0f", 1.0/elapsedTime);
00080 
00081  lightingOn= glIsEnabled(GL_LIGHTING);        /* lighting on? */
00082  if (lightingOn) glDisable(GL_LIGHTING);
00083 
00084  glGetIntegerv(GL_MATRIX_MODE, &matrixMode);  /* matrix mode? */
00085 
00086  glMatrixMode(GL_PROJECTION);
00087  glPushMatrix();
00088    glLoadIdentity();
00089    gluOrtho2D(0.0, 1.0, 0.0, 1.0);
00090    glMatrixMode(GL_MODELVIEW);
00091    glPushMatrix();
00092      glLoadIdentity();
00093      glPushAttrib(GL_COLOR_BUFFER_BIT);       /* save current colour */
00094        glColor3f(r, g, b);
00095        glRasterPos3f(x, y, 0.0);
00096        for(ch= str; *ch; ch++) {
00097           glutBitmapCharacter(font, (int)*ch);
00098        }
00099      glPopAttrib();
00100    glPopMatrix();
00101    glMatrixMode(GL_PROJECTION);
00102  glPopMatrix();
00103  glMatrixMode(matrixMode);
00104  if (lightingOn) glEnable(GL_LIGHTING);
00105 }
